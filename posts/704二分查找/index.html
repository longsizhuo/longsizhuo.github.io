704. 二分查找

https://leetcode.cn/leetbook/read/learning-algorithms-with-leetcode/xsshgi/ 不难，只是递归做个笔记

class Solution:
    def search(self, nums: List[int], target: int) -> int:
        def recursion(nums, left, right, target):
            if left > right:
                # 递归结束
                return -1
            mid = left + (right - left) // 2
            if nums[mid] == target:
                return mid
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
            return recursion(nums, left, right, target)

        left = 0
        right = len(nums) - 1
        return recursion(nums, left, right, target)

class Solution:
        def guessNumber(self, n: int) -> int:
            def recursion(nums, left, right):
                if left > right:
                    # 递归结束
                    return -1
                mid = left + (right - left) // 2
                if guess(mid) == 0:
                    return mid
                elif guess(mid) == 1:
                    left = mid + 1
                else:
                    right = mid - 1
                return recursion(nums, left, right)

            left = 1
            right = n
            return recursion(n, left, right)
