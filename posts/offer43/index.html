剑指 Offer 43. 1～n 整数中 1 出现的次数

class Solution:
    def countDigitOne(self, n: int) -> int:
        mulk = 1
        ans = 0
        while n >= mulk:
            ans += (n//(mulk * 10)) * mulk + min(max(n % (mulk * 10) - mulk + 1, 0), mulk)
            mulk *= 10
        return ans
